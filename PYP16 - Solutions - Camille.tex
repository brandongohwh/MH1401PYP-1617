def newton_sqrt(X,n):
    if X < 0 or n <= 0:
        return -1
    else:
        R1 = 10
        if n == 1:
            return R1
        else:
            R = newton_sqrt(X,n-1)
            Rn = 0.5*((R**2+X)/R)
            return Rn

def newton_sqrt_approx(X,a):
